using namespace std;

int main()
{

   cin>>n>>m;
   
   int arr[n][m];
   for(int i=0;i<n;i++){
       for(int j=0;j<m;j++){
           cin>>arr[i][j];
       }
   }

       for(int j=0;j<m;j++){
           cout<<arr[i][j];
       }

   }
   
   int x;
   cin>>x;
   
   bool flag=false;
   for(int i=0;i<n;i++){
       for(int j=0;j<m;j++){
           if(arr[i][j]==x){

           }
       }
   }
